  <script>
    const noBtn = document.getElementById('noBtn');
    const yesBtn = document.getElementById('yesBtn');
    const arena = document.getElementById('arena');
    const result = document.getElementById('result');
    const hearts = document.getElementById('hearts');

    function rand(min, max){ return Math.random() * (max - min) + min; }

    function moveNoButton(){
      const arenaRect = arena.getBoundingClientRect();
      const btnRect = noBtn.getBoundingClientRect();

      // limites pour rester dans la zone
      const maxX = arenaRect.width - btnRect.width;
      const maxY = arenaRect.height - btnRect.height;

      const x = rand(0, Math.max(0, maxX));
      const y = rand(0, Math.max(0, maxY));

      noBtn.style.left = `${x}px`;
      noBtn.style.top  = `${y}px`;
      noBtn.style.transform = `translate(0,0)`;
    }

    // Le bouton "Non" fuit d√®s qu'on approche / survol / touche
    ['mouseenter','mousemove','touchstart','pointerenter'].forEach(evt => {
      noBtn.addEventListener(evt, (e) => {
        // Emp√™che le clic sur mobile
        if (evt === 'touchstart') e.preventDefault();
        moveNoButton();
      }, { passive: false });
    });

    // Si malgr√© tout on clique "Non" (rare), on le fait fuir encore
    noBtn.addEventListener('click', (e) => {
      e.preventDefault();
      moveNoButton();
    });

    function spawnHeart(){
      const heart = document.createElement('div');
      heart.className = 'heart';
      heart.textContent = Math.random() < 0.5 ? 'üíñ' : 'üíò';
      heart.style.left = `${rand(0, 100)}%`;
      heart.style.bottom = `-20px`;
      heart.style.fontSize = `${rand(16, 28)}px`;
      heart.style.animationDuration = `${rand(2.5, 5)}s`;
      hearts.appendChild(heart);
      setTimeout(() => heart.remove(), 5500);
    }

    let heartTimer = null;

    yesBtn.addEventListener('click', () => {
      result.style.display = 'block';
      yesBtn.disabled = true;
      noBtn.disabled = true;
      yesBtn.style.filter = 'brightness(0.95)';
      noBtn.style.filter = 'brightness(0.8)';

      // pluie de coeurs
      spawnHeart();
      heartTimer = setInterval(spawnHeart, 180);

      // petit effet : on centre le "Oui"
      noBtn.style.display = 'none';
      arena.style.height = 'auto';
    });
  </script>
